/**
 * 上下文组装器
 *
 * 负责将当前章节相关的领域数据（角色锚点、历史摘要、近期原文）
 * 按 Token 预算拼装为 LLM 可用的上下文字符串。
 *
 * 设计约束：
 * - 总 Token 预算不超过 14500（Claude claude-sonnet-4-6 的安全上限）
 * - 锁定角色优先纳入，不受预算限制
 * - 各分区预算固定，见 {@link TokenBudget}
 *
 * @author wenshu-team
 * @since 1.0.0
 */
@Service
@Slf4j
public class ContextAssembler {

    /** 历史摘要最多纳入最近 N 章 */
    private static final int MAX_SUMMARY_CHAPTERS = 3;

    /** 近期原文滑动窗口字符数（约 1500 汉字 ≈ 2500 token） */
    private static final int RECENT_CONTEXT_CHARS = 1500;

    private final ChapterRepository chapterRepository;
    private final CharacterRepository characterRepository;
    private final ChapterSummaryRepository summaryRepository;

    /**
     * 为指定章节组装 LLM 上下文。
     *
     * @param chapterId 目标章节 ID
     * @param taskType  当前任务类型，影响各分区权重分配
     * @return 包含各分区内容的上下文对象，保证总 Token 不超预算
     * @throws ChapterNotFoundException 若章节不存在
     */
    public AssembledContext assemble(UUID chapterId, TaskType taskType) {
        log.info("[ContextAssembler] 开始组装上下文, chapterId={}, taskType={}", chapterId, taskType);

        Chapter chapter = chapterRepository.findById(chapterId)
            .orElseThrow(() -> new ChapterNotFoundException(chapterId));

        // 1. 锁定角色：优先纳入，不受预算压缩
        List<CharacterAnchor> lockedAnchors = characterRepository
            .findLockedByProjectId(chapter.getProjectId());
        log.debug("[ContextAssembler] 锁定角色数量={}", lockedAnchors.size());

        // 2. 按重要性评分填入剩余预算
        List<CharacterAnchor> scoredAnchors = characterRepository
            .findByProjectId(chapter.getProjectId());
        List<CharacterAnchor> selectedAnchors = anchorSelector
            .selectByBudget(scoredAnchors, chapter.getContent(), TokenBudget.ANCHORS);

        // 3. 前 N 章压缩摘要
        String historySummary = summaryRepository
            .findRecentByProjectId(chapter.getProjectId(), chapterId, MAX_SUMMARY_CHAPTERS)
            .stream()
            .map(ChapterSummary::getSummary)
            .collect(Collectors.joining("\n"));

        // 4. 最近原文滑动窗口
        String recentContext = chapterRepository
            .findRecentContentBefore(chapter.getProjectId(), chapterId, RECENT_CONTEXT_CHARS);

        AssembledContext context = AssembledContext.builder()
            .anchorCharacters(formatAnchors(lockedAnchors, selectedAnchors))
            .worldviewSummary(chapter.getProject().getWorldview())
            .historySummary(historySummary)
            .recentContext(recentContext)
            .chapterOutline(chapter.getOutline())
            .build();

        log.info("[ContextAssembler] 上下文组装完成, 预估Token={}", context.estimateTokens());
        return context;
    }
}

0.3 设计模式应用

模式应用位置解决的问题
策略模式（Strategy）SceneExternalizationStrategy 接口 + 四个实现类心理外化的四种策略（动作/对白/独白/跳过）可独立扩展，避免 if-else 堆叠
工厂模式（Factory）LlmClientFactory根据任务类型（创意/分类/审查）返回对应 LLM 客户端，调用方无需感知具体模型
模板方法（Template Method）AbstractPolishHandler三级润色（基础/进阶/风格）共享预处理和后处理步骤，只有核心 Prompt 不同
建造者模式（Builder）AssembledContext.Builder、SkeletonResult.Builder构造复杂对象时，避免过多构造器参数，提高可读性
观察者模式（Observer）ChapterCompletedEvent + Spring 事件机制章节完成后异步触发锚点提取、摘要生成，解耦主流程与后台任务
适配器模式（Adapter）ClaudeChatAdapter、DeepSeekChatAdapter统一封装不同 LLM 的调用差异，上层代码只依赖 ChatAdapter 接口
责任链（Chain of Responsibility）LlmFallbackChain模型降级：Claude 失败 → DeepSeek → 抛出异常，每个节点只处理自己的情况

策略模式示例（心理外化）

/**
 * 心理外化策略接口
 *
 * 每种策略负责将小说中的心理描写转化为剧本可表现的形式。
 * 新增策略只需实现此接口，无需修改已有代码（开闭原则）。
 */
public interface SceneExternalizationStrategy {

    /**
     * 返回策略唯一标识，与前端传参对应。
     * 示例：ACTION, DIALOGUE, VOICEOVER, SKIP
     */
    String getStrategyCode();

    /**
     * 返回注入 Prompt 的策略指令文本。
     *
     * @return 策略指令，用于替换 Prompt 中的 {{strategy_instruction}} 占位符
     */
    String getInstructionText();
}

/** 动作外化策略：将内心活动转为可观察的身体动作 */
@Component
public class ActionExternalizationStrategy implements SceneExternalizationStrategy {

    @Override
    public String getStrategyCode() {
        return "ACTION";
    }

    @Override
    public String getInstructionText() {
        return "将人物内心活动和情绪状态，转化为可被摄影机捕捉的身体动作和行为细节。"
             + "示例：'他感到绝望' → '他慢慢蹲下，双手捂脸，肩膀细微颤抖'";
    }
}

/**
 * 策略工厂：根据前端传入的 strategyCode 返回对应策略实例。
 * 所有策略 Bean 由 Spring 自动注入，新增策略无需修改此类。
 */
@Component

    private static final Logger log = LoggerFactory.getLogger(ExternalizationStrategyFactory.class);

    /** key: strategyCode, value: 策略实现 */
    private final Map<String, SceneExternalizationStrategy> strategyMap;

    public ExternalizationStrategyFactory(List<SceneExternalizationStrategy> strategies) {
        this.strategyMap = strategies.stream()
            .collect(Collectors.toMap(
                SceneExternalizationStrategy::getStrategyCode,
                Function.identity()
            ));
        log.info("[ExternalizationStrategyFactory] 已加载策略数量={}", strategyMap.size());
    }

    /**
     * 根据策略代码获取对应策略。
     *
     * @param code 策略代码（如 ACTION/DIALOGUE/VOICEOVER/SKIP）
     * @return 对应策略实现
     * @throws IllegalArgumentException 若策略代码不存在
     */
    public SceneExternalizationStrategy getStrategy(String code) {
        SceneExternalizationStrategy strategy = strategyMap.get(code);
        if (strategy == null) {
            log.error("[ExternalizationStrategyFactory] 未知策略代码, code={}", code);
            throw new IllegalArgumentException("未知的心理外化策略: " + code);
        }
        return strategy;
    }

JsonParseUtil — JSON 解析健壮性工具（common/util/JsonParseUtil.java）

/**
 * LLM JSON 响应解析工具
 *
 * 处理以下常见的 LLM JSON 输出问题：
 * 1. Markdown fence 包裹（```json ... ```）
 * 2. 输出截断（JSON 不完整，补全末尾括号）
 * 3. 中文引号导致的 JSON 损坏（"内容" → \"内容\"）
 * 4. 尾随逗号（trailing comma）
 */
@UtilityClass
public class JsonParseUtil {

    private static final ObjectMapper MAPPER = new ObjectMapper()
        .configure(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false);

    /**
     * 带自动修复的 JSON 解析，修复失败则执行 fallback supplier。
     */
    public static <T> T parseWithFallback(String raw, Class<T> type,
                                           Supplier<T> fallback) {
        try {
            return strictParse(autoRepair(raw), type);
        } catch (Exception e) {
            log.warn("[JsonParseUtil] 自动修复失败，执行 fallback, error={}", e.getMessage());
            return fallback.get();
        }
    }

    public static <T> T strictParse(String json, Class<T> type) throws JsonProcessingException {
        return MAPPER.readValue(json, type);
    }

    /** 自动修复常见 JSON 格式问题 */
    static String autoRepair(String raw) {
        String s = raw.trim();
        // 1. 去除 markdown fence
        if (s.startsWith("```")) {
            s = s.replaceAll("^```[a-zA-Z]*\\n?", "").replaceAll("```$", "").trim();
        }
        // 2. 去除尾随逗号（,} 或 ,]）
        s = s.replaceAll(",\\s*([}\\]])", "$1");
        // 3. 截断修复：找到最后一个完整的 } 或 ]
        if (!isValid(s)) {
            s = truncationRepair(s);
        }
        return s;
    }

    private static boolean isValid(String json) {
        try { MAPPER.readTree(json); return true; }
        catch (Exception e) { return false; }
    }

    private static String truncationRepair(String json) {
        // 计算未闭合的括号数量，补全到结尾
        int curly = 0, square = 0;
        for (char c : json.toCharArray()) {
            if (c == '{') curly++; else if (c == '}') curly--;
            if (c == '[') square++; else if (c == ']') square--;
        }
        StringBuilder sb = new StringBuilder(json);
        for (int i = 0; i < square; i++) sb.append(']');
        for (int i = 0; i < curly; i++) sb.append('}');
        return sb.toString();
    }
}

        int claudeCalls = 0;
        for (ConsistencyIssue issue : dedupedIssues) {
            if (issue.getConfidence() < 0.7 && claudeCalls < MAX_CLAUDE_VERIFY_CALLS) {
                boolean confirmed = claudeVerify(issue);
                issue.setVerified(confirmed);
                claudeCalls++;
                if (!confirmed) issue.setStatus("false_positive");
            } else {
                issue.setVerified(true); // 置信度高，无需二次确认
            }
        }

        log.info("[ConsistencyService] 审查完成, claudeVerifyCalls={}/{}", claudeCalls, MAX_CLAUDE_VERIFY_CALLS);

        // 5. 过滤误报，存库
        List<ConsistencyIssue> finalIssues = dedupedIssues.stream()
            .filter(i -> !"false_positive".equals(i.getStatus()))
            .collect(toList());

        return saveReport(projectId, finalIssues);
    }
}

@Service
public class ScriptConversionService {

    @Async("scriptExecutor")
    public CompletableFuture<ScriptDraft> convert(ScriptConversionRequest req) {
        // 1. 预处理：超过 5000 字则分批（每批 3000 字，保留 200 字重叠）
        List<String> batches = TextChunker.chunk(req.getNovelText(), 3000, 200);

        // 2. 场景分割（DeepSeek，低成本）
        List<SceneUnit> scenes = batches.stream()
            .flatMap(b -> sceneSegmentationService.segment(b).stream())
            .collect(toList());

        // 3. 逐场改编
        //    importance >= 0.7 的关键场景：串行，注入前场结尾
        //    importance < 0.7 的过渡场景：并行，最多 5 个同时进行
        List<SceneScript> scripts = adaptScenes(scenes, req.getStrategy());

        // 4. 格式校验 + 修复
        scripts = formatValidator.validateAndFix(scripts);

        // 5. 组装完整剧本并存库
        ScriptDraft draft = assembleAndSave(scripts, req);
        return CompletableFuture.completedFuture(draft);
    }

    private List<SceneScript> adaptScenes(List<SceneUnit> scenes, ExternalizationStrategy strategy) {
        List<SceneScript> result = new ArrayList<>();
        List<SceneUnit> parallel = scenes.stream().filter(s -> s.getImportance() < 0.7).toList();
        List<SceneUnit> serial   = scenes.stream().filter(s -> s.getImportance() >= 0.7).toList();

        // 并行处理低重要度场景
        List<CompletableFuture<SceneScript>> futures = parallel.stream()
            .map(s -> CompletableFuture.supplyAsync(() -> adaptSingleScene(s, strategy, null)))
            .toList();
        result.addAll(futures.stream().map(CompletableFuture::join).toList());

        // 串行处理关键场景（注入前场结尾）
        String prevEnding = "";
        for (SceneUnit scene : serial) {
            SceneScript adapted = adaptSingleScene(scene, strategy, prevEnding);
            prevEnding = extractLastParagraph(adapted.getContent());
            result.add(adapted);
        }

        result.sort(Comparator.comparingInt(SceneScript::getSceneIndex));
        return result;
    }
}

3.3 重要性评分算法

public int scoreAnchorImportance(Character c, String currentText) {
    int score = 0;
    // 关键词命中：角色名在当前章节出现频率（最高 30 分）
    long nameCount = countOccurrences(currentText, c.getName());
    score += Math.min(30, nameCount * 5);
    // 锁定加成（20 分）
    if (c.isLocked()) score += 20;
    // 近期活跃度：最近 3 章出现过（15 分）
    if (chapterRepo.appearsInRecentChapters(c.getId(), 3)) score += 15;
    // 角色重要性：主角固定加分（主角 20 / 配角 10 / 龙套 0）
    score += switch (c.getRole()) { case "主角" -> 20; case "配角" -> 10; default -> 0; };
    return score; // 总分 0–85
}

8.3 SSE 流式续写接口

@RestController
@RequestMapping("/api/v1/novel")
public class NovelController {

    @GetMapping(value = "/continue", produces = MediaType.TEXT_EVENT_STREAM_VALUE)
    public SseEmitter continueWriting(
            @RequestParam UUID chapterId,
            @RequestParam(required = false) String instruction,
            @AuthenticationPrincipal UserDetails user) {

        SseEmitter emitter = new SseEmitter(60_000L); // 60s total timeout

        // 首字超时检测：8秒内未收到第一个 token，主动提示用户
        AtomicBoolean firstTokenReceived = new AtomicBoolean(false);
        ScheduledFuture firstTokenTimer = scheduler.schedule(() -> {
            if (!firstTokenReceived.get()) {
                try {
                    // 推送特殊事件通知前端
                    emitter.send(SseEmitter.event()
                        .name("FIRST_TOKEN_TIMEOUT")
                        .data("{\"message\":\"响应较慢，服务器仍在处理，可继续等待或取消\"}"));
                    log.warn("[NovelController] 首字超时 8s, chapterId={}", chapterId);
                } catch (IOException e) {
                    emitter.completeWithError(e);
                }
            }
        }, 8, TimeUnit.SECONDS);

        // 组装上下文
        AssembledContext ctx = contextAssembler.assemble(chapterId, TaskType.CONTINUATION);
        String prompt = promptFiller.fill(ContinuationPrompts.CONTINUATION, ctx, instruction);

        // 配额检查
        quotaService.checkAndDeduct(user.getUserId(), EstimatedTokens.CONTINUATION);

        // 流式调用
        claudeStreamingModel.generate(prompt, new StreamingResponseHandler<AiMessage>() {
            @Override public void onNext(String token) {
                firstTokenReceived.compareAndSet(false, true);
                firstTokenTimer.cancel(false);
                try { emitter.send(SseEmitter.event().data(token)); }
                catch (IOException e) { emitter.completeWithError(e); }
            }
            @Override public void onComplete(Response<AiMessage> r) {
                // 异步触发：锚点提取 + 章节摘要
                anchorExtractionService.extractAsync(chapterId);
                emitter.complete();
            }
            @Override public void onError(Throwable e) { emitter.completeWithError(e); }
        });

        return emitter;
    }
}

10.1 认证与用户接口

登录体系为邮箱 + 密码，注册后需邮箱验证方可使用 AI 功能。双 Token 策略：Access Token（JWT，15 分钟）+ Refresh Token（随机字符串，30 天，存 Redis）。所有认证接口均不需要 Bearer Token，其余接口必须携带。

POST/auth/register邮箱 + 密码注册（发送激活邮件，自动登录，返回双 Token）否

POST/auth/resend-verify重新发送验证邮件（限流：60s/次）否
POST/auth/login邮箱 + 密码登录（连续错误 5 次锁定 15 分钟）否
POST/auth/password/forgot发起密码重置（发送重置邮件，含 UUID token，1 小时有效）否
POST/auth/password/reset重置密码（携带 UUID token + 新密码，成功后吊销所有 Refresh Token）否
POST/auth/refresh静默刷新 Access Token（携带 Refresh Token，轮换后旧 Token 立即失效）否

/**
 * 编辑场景内容（乐观锁保护）
 *
 * @param sceneId       场景 ID
 * @param content       新内容
 * @param clientVersion 客户端持有的版本号
 * @throws OptimisticLockException 版本不匹配时抛出，由 GlobalExceptionHandler 转为 409
 */
@Transactional
public SceneVersionDto editScene(UUID sceneId, String content, int clientVersion) {
    // 1. 尝试 CAS 更新
    int affected = scriptSceneMapper.updateWithVersion(sceneId, content, clientVersion);

    if (affected == 0) {
        // 2. 版本不匹配：读取最新状态供前端对比
        ScriptScenePO latest = scriptSceneMapper.findById(sceneId)
            .orElseThrow(() -> new ResourceNotFoundException("场景不存在: " + sceneId));
        log.warn("[ScriptSceneService] 乐观锁冲突, sceneId={}, clientVersion={}, latestVersion={}",
            sceneId, clientVersion, latest.getVersion());
        throw new OptimisticLockException(latest.getVersion(), latest.getContent());
    }

    return SceneVersionDto.of(sceneId, clientVersion + 1);
}

// composables/useSceneEditor.ts
// 规则：每次保存成功后更新本地 version；409 时弹出 diff 对话框

async function saveScene(sceneId: string, content: string) {
  try {
    const res = await scriptApi.updateScene(sceneId, {
      content,
      version: localVersion.value  // 本地持有的版本号
    })
    localVersion.value = res.data.version  // 保存成功，刷新版本号
    isDirty.value = false
  } catch (err: any) {
    if (err.response?.status === 409) {
      const { latestVersion, latestContent } = err.response.data.data
      // 弹出冲突处理对话框：让用户选择「保留我的版本」或「使用最新版本」
      const choice = await showConflictDialog({
        myContent: content,
        theirContent: latestContent
      })
      if (choice === 'mine') {
        // 强制覆盖：用最新版本号重发请求
        localVersion.value = latestVersion
        await saveScene(sceneId, content)  // 递归重试一次
      } else {
        // 丢弃本地修改，使用服务端最新内容
        localContent.value  = latestContent
        localVersion.value = latestVersion
        isDirty.value = false
      }
    } else {
      throw err
    }
  }
}

11.3 Java 进度上报实现

/**
 * 任务进度更新服务（供各 @Async 方法调用）。
 * 通过更新 ai_task_progress 表，前端轮询时即可获取最新进度。
 */
@Service
@Slf4j
public class TaskProgressService {

    private final AiTaskProgressMapper progressMapper;

    /**
     * 初始化任务进度记录（任务启动时调用）。
     *
     * @param taskId    任务唯一标识（UUID）
     * @param taskType  任务类型
     * @param totalSteps 总步骤数
     */
    public void init(UUID taskId, String taskType, int totalSteps) {
        AiTaskProgressPO po = new AiTaskProgressPO();
        po.setId(taskId);
        po.setTaskType(taskType);
        po.setStatus("running");
        po.setTotalSteps(totalSteps);
        po.setCurrentStep(0);
        po.setProgressPct(0);
        progressMapper.insert(po);
        log.info("[TaskProgress] 任务开始, taskId={}, type={}, totalSteps={}", taskId, taskType, totalSteps);
    }

    /**
     * 更新当前步骤进度（每完成一个步骤后调用）。
     *
     * @param taskId     任务 ID
     * @param step       当前完成的步骤序号（从 1 开始）
     * @param stepLabel  展示给用户的步骤描述
     */
    public void step(UUID taskId, int step, String stepLabel) {
        progressMapper.updateStep(taskId, step, stepLabel,
            calculatePct(step, progressMapper.getTotalSteps(taskId)));
        log.debug("[TaskProgress] 步骤更新, taskId={}, step={}, label={}", taskId, step, stepLabel);
    }

    /**
     * 标记任务完成，记录结果 ID。
     *
     * @param taskId   任务 ID
     * @param resultId 业务结果 ID（draftId / reportId 等）
     */
    public void complete(UUID taskId, UUID resultId) {
        progressMapper.markCompleted(taskId, resultId);
        log.info("[TaskProgress] 任务完成, taskId={}, resultId={}", taskId, resultId);
    }

    /**
     * 标记任务失败，记录错误信息。
     *
     * @param taskId  任务 ID
     * @param message 对用户可读的错误描述
     */
    public void fail(UUID taskId, String message) {
        progressMapper.markFailed(taskId, message);
        log.error("[TaskProgress] 任务失败, taskId={}, error={}", taskId, message);
    }

    private int calculatePct(int current, int total) {
        return total == 0 ? 0 : Math.min(100, (int) (current * 100.0 / total));

11.4 接口安全与防滥用

  • 任务归属校验：查询进度时验证 taskId 属于当前登录用户，防止枚举攻击
  • 轮询限流：同一 taskId 每秒最多查询 2 次（Redis 计数），超出返回 429
  • 过期清理：任务完成或失败后保留 24 小时，之后由定时任务清理
  • 超时兜底：任务运行超过 5 分钟自动标记为 failed，前端展示「生成超时，请重试」

// 键格式：consistency:quota:{userId}:{yearMonth}
// 值：当月已使用次数（整数）
// TTL：当月最后一天 23:59:59 自动过期

/**
 * 一致性审查配额检查服务
 * 调用时机：触发审查任务前，在 ConsistencyCheckAppService 中调用
 */
@Service
@RequiredArgsConstructor
public class ConsistencyQuotaService {

    private static final Map<String, Integer> MONTHLY_LIMIT = Map.of(
        "free",   1,
        "pro",    5,
        "master", 20,
        "team",   Integer.MAX_VALUE   // 不限
    );

    private static final Map<String, Long> WORD_LIMIT = Map.of(
        "free",   20_000L,
        "pro",    500_000L,
        "master", 2_000_000L,
        "team",   Long.MAX_VALUE
    );

    private final RedisTemplate<String, String> redis;
    private final UserRepository userRepository;

    /**
     * 检查审查配额，通过则扣减，不通过则抛出业务异常
     *
     * @param userId       用户 ID
     * @param novelWords   本次审查的总字数
     */
    public void checkAndConsume(UUID userId, long novelWords) {
        User user = userRepository.findById(userId).orElseThrow();
        String plan = user.getPlan();

        // 1. 字数上限校验
        long wordLimit = WORD_LIMIT.getOrDefault(plan, 20_000L);
        if (novelWords > wordLimit) {
            throw new QuotaExceededException(
                String.format("当前套餐单次审查上限为 %,d 字，你的作品共 %,d 字，请升级套餐",
                    wordLimit, novelWords));
        }

        // 2. 月次数校验（team 直接跳过）
        int monthlyLimit = MONTHLY_LIMIT.getOrDefault(plan, 1);
        if (monthlyLimit == Integer.MAX_VALUE) return;

        String key = buildKey(userId);
        String val  = redis.opsForValue().get(key);
        int used = val == null ? 0 : Integer.parseInt(val);

        if (used >= monthlyLimit) {
            throw new QuotaExceededException(
                String.format("本月一致性审查次数已用完（%d/%d），下月自动重置", used, monthlyLimit));
        }

        // 3. 原子扣减，TTL 设置到当月末
        redis.opsForValue().increment(key);
        if (used == 0) {
            // 首次写入，设置到当月最后一秒
            redis.expireAt(key, endOfMonth());
        }
        log.info("[ConsistencyQuota] 审查次数扣减, userId={}, plan={}, used={}/{}",
            userId, plan, used + 1, monthlyLimit);
    }

    /** 查询当月已用次数（用于前端配额展示） */
    public int getUsed(UUID userId) {
        String val = redis.opsForValue().get(buildKey(userId));
        return val == null ? 0 : Integer.parseInt(val);
    }

    private String buildKey(UUID userId) {
        String yearMonth = LocalDate.now().format(DateTimeFormatter.ofPattern("yyyy-MM"));
        return "consistency:quota:" + userId + ":" + yearMonth;
    }

    private Instant endOfMonth() {
        return LocalDate.now()
            .with(TemporalAdjusters.lastDayOfMonth())
            .atTime(23, 59, 59)
            .toInstant(ZoneOffset.UTC);
    }
}

quota store 必须是全局单例，被首页进度条、超限拦截弹窗、Tooltip 共享，任何本地复制都会导致状态不同步。

// stores/quota.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { getUserQuota } from '@/api/user'
import type { QuotaDetail } from '@/types/user'

export const useQuotaStore = defineStore('quota', () => {
  const detail = ref<QuotaDetail | null>(null)

  /** 刷新配额详情（登录成功 / 每次 AI 操作完成后调用） */
  async function refresh() {
    detail.value = await getUserQuota()
  }

  /**
   * 检查一致性审查配额，返回拦截级别：
   *   block  — 免费版次数用完，弹升级弹窗，阻止操作
   *   toast  — 专业版/大师版次数用完，Toast 提示，操作仍继续
   *   word_limit — 字数超出套餐上限，弹范围选择弹窗
   *   ok     — 配额充足
   */
  const consistencyCheckLevel = computed(() => {
    if (!detail.value) return 'ok'
    const { plan, usedConsistency, quotaConsistency } = detail.value
    if (usedConsistency >= quotaConsistency) {
      return plan === 'free' ? 'block' : 'toast'
    }
    return 'ok'
  })

  /**
   * 检查剧本改编次数配额
   *   block  — 免费版 1/1 用完，弹升级弹窗
   *   upsell — 付费版次数用完，弹升级到更高套餐弹窗
   *   ok     — 配额充足
   */
  const adaptationCheckLevel = computed(() => {
    if (!detail.value) return 'ok'
    const { plan, usedAdaptations, quotaAdaptations } = detail.value
    if (usedAdaptations >= quotaAdaptations) {
      return plan === 'free' ? 'block' : 'upsell'
    }
    return 'ok'
  })

  return { detail, refresh, consistencyCheckLevel, adaptationCheckLevel }
})

task store — 统一任务轮询管理

骨架生成、剧本改编、一致性审查、数据导出均为异步任务，统一在 task store 管理生命周期，防止切换页面后轮询泄漏。

// stores/task.ts
import { defineStore } from 'pinia'
import { ref } from 'vue'
import { getTaskProgress } from '@/api/task'

type TaskType = 'skeleton_generation' | 'script_conversion' | 'consistency_check' | 'data_export'
type TaskStatus = 'pending' | 'running' | 'completed' | 'failed'

/** 每种任务类型的轮询间隔（毫秒） */
const POLL_INTERVAL: Record<TaskType, number> = {
  skeleton_generation: 2000,
  script_conversion:   3000,
  consistency_check:   5000,
  data_export:         2000,
}

interface TaskState {
  taskId:    string
  type:      TaskType
  status:    TaskStatus
  progress:  number      // 0–100
  stepLabel: string
  resultId?: string
  timerId?:  ReturnType<typeof setInterval>
}

export const useTaskStore = defineStore('task', () => {
  /** key: taskId */
  const tasks = ref<Map<string, TaskState>>(new Map())

  /**
   * 启动任务轮询。若该 taskId 已在轮询中则直接返回（防重复）。
   * @param taskId  服务端返回的任务 ID
   * @param type    任务类型（决定轮询间隔）
   */
  function startPolling(taskId: string, type: TaskType) {
    if (tasks.value.has(taskId)) return

    const state = ref<TaskState>({ taskId, type, status: 'pending', progress: 0, stepLabel: '' })
    tasks.value.set(taskId, state.value)

    const interval = POLL_INTERVAL[type]
    const timer = setInterval(async () => {
      try {
        const res = await getTaskProgress(taskId)
        Object.assign(state.value, res)
        if (res.status === 'completed' || res.status === 'failed') {
          clearInterval(timer)  // 任务终态，停止轮询
        }
      } catch {
        // 网络抖动不停止轮询，服务端超时兜底（5分钟后自动 failed）
      }
    }, interval)

    state.value.timerId = timer
  }

  /** 手动停止指定任务的轮询（用户点击「取消改编」等场景） */
  function stopPolling(taskId: string) {
    const state = tasks.value.get(taskId)
    if (state?.timerId) {
      clearInterval(state.timerId)
      tasks.value.delete(taskId)
    }
  }

  /** 页面卸载时清理全部轮询，防止内存泄漏 */
  function disposeAll() {
    tasks.value.forEach(s => s.timerId && clearInterval(s.timerId))
    tasks.value.clear()
  }

  return { tasks, startPolling, stopPolling, disposeAll }
})

SSE 统一封装

续写、骨架生成进度、剧本改编进度三个场景均使用 SSE，统一封装避免重复逻辑、防止连接泄漏。

注意：SSE 不能携带 Authorization Header，Access Token 必须通过 URL query 参数传递。

// composables/useSseStream.ts
import { ref, onUnmounted } from 'vue'
import { useAuthStore } from '@/stores/auth'

interface SseOptions {
  /** SSE 端点 URL（不含 token 参数） */
  url: string
  /** 收到每个 token 时的回调 */
  onToken: (token: string) => void
  /** 流式输出正常完成时的回调 */
  onComplete: () => void
  /** 出错或重试耗尽时的回调 */
  onError: (err: Error) => void
  /** 最大重试次数，默认 3 */
  maxRetry?: number
}

/**
 * 统一 SSE 连接管理 composable
 *
 * 功能：
 *  - 自动在 URL 中注入最新 Access Token（query: ?token=...）
 *  - 指数退避重连（1s、2s、4s），超过 maxRetry 次后调用 onError
 *  - 组件卸载时自动关闭连接，防止内存泄漏
 *  - 同一 composable 实例同时只保持一个活跃连接（startStream 自动关闭旧连接）
 */
export function useSseStream(options: SseOptions) {
  const { url, onToken, onComplete, onError, maxRetry = 3 } = options
  const authStore = useAuthStore()

  const isStreaming = ref(false)
  let source: EventSource | null = null
  let retryCount = 0

  function connect(extraParams: Record<string, string> = {}) {
    // SSE 必须用 query param 传 token，不能用 Header
    const params = new URLSearchParams({
      ...extraParams,
      token: authStore.accessToken
    })
    source = new EventSource(`${url}?${params}`)
    isStreaming.value = true

    source.onmessage = (e) => {
      if (e.data === '[DONE]') { close(); onComplete(); return }
      onToken(e.data)
    }

    source.onerror = () => {
      close()
      if (retryCount < maxRetry) {
        const delay = Math.pow(2, retryCount) * 1000  // 指数退避：1s, 2s, 4s
        retryCount++
        setTimeout(() => connect(extraParams), delay)
      } else {
        onError(new Error(`SSE 连接失败，已重试 ${maxRetry} 次`))
      }
    }
  }

  function startStream(params: Record<string, string> = {}) {
    close()          // 先关闭旧连接
    retryCount = 0   // 重置重试计数
    connect(params)
  }

  function close() {
    source?.close()
    source = null
    isStreaming.value = false
  }

  // 组件卸载时自动清理，防止 SSE 连接在后台持续运行
  onUnmounted(close)

  return { isStreaming, startStream, close }
}

TipTap + SSE 集成：RAF 批量写入

每个 SSE token 触发一次 ProseMirror transaction 代价极高（重绘 + 历史记录），必须通过 requestAnimationFrame 缓冲合并写入。

// composables/useTiptapSse.ts
import type { Editor } from '@tiptap/vue-3'

/**
 * TipTap 编辑器的 SSE Token 批量写入器
 *
 * 工作原理：
 *  每个 token 写入 buffer；在下一个 animation frame 统一 insertContent。
 *  同一帧内的多个 token 合并为一次 transaction，大幅减少 ProseMirror 重绘。
 *
 * @param getEditor 获取 TipTap editor 实例的函数（响应式 ref 场景用 () => editor.value）
 */
export function useTiptapSse(getEditor: () => Editor | null) {
  let buffer = ''
  let rafId: number | null = null

  /** 接收一个 SSE token，缓冲后在下一帧写入编辑器 */
  function onToken(token: string) {
    buffer += token
    if (rafId === null) {
      rafId = requestAnimationFrame(flush)
    }
  }

  /** 将缓冲区内容写入编辑器，一帧一次 transaction */
  function flush() {
    rafId = null
    const editor = getEditor()
    if (!editor || !buffer) return
    editor.commands.insertContent(buffer)
    buffer = ''
  }

  /** 流式完成时调用，强制刷新剩余 buffer */
  function complete() {
    if (rafId !== null) {
      cancelAnimationFrame(rafId)
      rafId = null
    }
    flush()
  }

  return { onToken, complete }
}

API 层：Token 静默刷新（队列锁）

多个并发请求同时触发 401 时，若不加锁会导致 Refresh Token 被消费多次（轮换机制下旧 Token 立即失效，后续请求会用失效 Token 刷新失败）。队列锁确保同一时刻只有一次刷新，其余请求排队等待。

// api/interceptors/auth.ts
import type { AxiosInstance, InternalAxiosRequestConfig } from 'axios'
import { refreshToken } from '@/api/auth'
import { useAuthStore } from '@/stores/auth'

/** 等待刷新的请求队列 */
let isRefreshing = false
let pendingQueue: Array<{ resolve: (token: string) => void; reject: (err: unknown) => void }> = []

export function setupAuthInterceptor(http: AxiosInstance) {
  http.interceptors.response.use(null, async (error) => {
    const status = error.response?.status
    const config = error.config as InternalAxiosRequestConfig & { _retried?: boolean }

    // 只处理 401，且不重复处理已重试过的请求
    if (status !== 401 || config._retried) return Promise.reject(error)
    config._retried = true

    if (isRefreshing) {
      // 其他请求正在刷新，排队等待新 Token
      return new Promise((resolve, reject) => {
        pendingQueue.push({
          resolve: (token) => { config.headers.Authorization = `Bearer ${token}`; resolve(http(config)) },
          reject
        })
      })
    }

    isRefreshing = true
    const authStore = useAuthStore()

    try {
      const { accessToken, refreshToken: newRefreshToken } = await refreshToken(authStore.refreshToken)
      authStore.saveTokens(accessToken, newRefreshToken)

      // 唤醒所有排队请求，注入新 Token
      pendingQueue.forEach(p => p.resolve(accessToken))

      config.headers.Authorization = `Bearer ${accessToken}`
      return http(config)
    } catch (refreshError) {
      // 刷新失败（Refresh Token 过期）→ 全部拒绝，跳转登录页
      pendingQueue.forEach(p => p.reject(refreshError))
      authStore.logout()
      return Promise.reject(refreshError)
    } finally {
      isRefreshing = false
      pendingQueue = []
    }
  })
}

多标签页检测：心跳机制

用于剧本工作台检测同一场景是否在其他标签页被同时编辑（IDD §7.6），页面卸载时必须清理心跳 key，否则残留 key 会导致永久误报「另一标签页正在编辑」。

// composables/useTabHeartbeat.ts
import { onUnmounted } from 'vue'

const TAB_KEY = 'wenshu_scene_editing'
const BEAT_INTERVAL = 2000  // 每 2 秒写一次心跳
const STALE_MS = 5000       // 超过 5 秒未更新视为失活

interface HeartbeatData {
  tabId:   string
  sceneId: string
  ts:      number
}

/**
 * 多标签页编辑检测（基于 localStorage 心跳）
 *
 * @param sceneId 当前正在编辑的场景 ID
 * @returns isOtherTabEditing — 检查当前是否有其他标签页正在编辑同一场景
 */
export function useTabHeartbeat(sceneId: string) {
  // 每个标签页的唯一 ID，sessionStorage 保证刷新后不变
  const tabId = sessionStorage.getItem('wenshu_tab_id') ?? (() => {
    const id = crypto.randomUUID()
    sessionStorage.setItem('wenshu_tab_id', id)
    return id
  })()

  function beat() {
    const data: HeartbeatData = { tabId, sceneId, ts: Date.now() }
    localStorage.setItem(TAB_KEY, JSON.stringify(data))
  }

  function isOtherTabEditing(): boolean {
    const raw = localStorage.getItem(TAB_KEY)
    if (!raw) return false
    try {
      const data: HeartbeatData = JSON.parse(raw)
      return data.tabId !== tabId
        && data.sceneId === sceneId
        && Date.now() - data.ts < STALE_MS
    } catch { return false }
  }

  const timer = setInterval(beat, BEAT_INTERVAL)
  beat()  // 立即写入，宣告占用

  // 关键：卸载时必须清理，否则心跳 key 永久残留导致永久误报
  function cleanup() {
    clearInterval(timer)
    // 只清除属于本标签页的 key，不影响其他标签页
    const raw = localStorage.getItem(TAB_KEY)
    if (raw) {
      try {
        const data: HeartbeatData = JSON.parse(raw)
        if (data.tabId === tabId) localStorage.removeItem(TAB_KEY)
      } catch { /* ignore */ }
    }
  }

  window.addEventListener('beforeunload', cleanup)
  onUnmounted(() => { cleanup(); window.removeEventListener('beforeunload', cleanup) })

  return { isOtherTabEditing }
}

ScriptDraft Store — Scene Version 并发状态设计

剧本工作台的每个场景需要独立维护版本号，支持乐观锁冲突检测：

// stores/scriptDraft.ts

interface SceneState {
  content: string
  version: number       // 服务端当前版本号
  dirty: boolean        // 本地有未保存改动
  saveStatus: 'idle' | 'saving' | 'conflict' | 'conflict_dirty' | 'error'
  conflictData?: {
    latestVersion: number
    latestContent: string
  }
}

export const useScriptDraftStore = defineStore('scriptDraft', () => {
  // Map<sceneId, SceneState> — 每个场景独立维护状态
  const scenes = ref<Map<string, SceneState>>(new Map())

  /** 从服务端加载场景列表，初始化版本号 */
  async function loadScenes(draftId: string) {
    const res = await api.getScenes(draftId)
    res.data.scenes.forEach(s => {
      scenes.value.set(s.id, {
        content: s.content,
        version: s.version,
        dirty: false,
        saveStatus: 'idle'
      })
    })
  }

  /** 用户编辑内容 — 标记 dirty */
  function updateContent(sceneId: string, content: string) {
    const scene = scenes.value.get(sceneId)
    if (!scene) return
    scene.content = content
    scene.dirty = true
  }

  /** 保存场景（带乐观锁，debounce 3s 后调用） */
  async function saveScene(sceneId: string) {
    const scene = scenes.value.get(sceneId)
    if (!scene || !scene.dirty) return
    if (scene.saveStatus === 'saving' || scene.saveStatus === 'conflict') return // guard

    scene.saveStatus = 'saving'
    try {
      const res = await api.putScene(sceneId, {
        content: scene.content,
        version: scene.version  // 携带客户端版本号
      })
      scene.version = res.data.version  // 更新为服务端最新版本号
      scene.dirty = false
      scene.saveStatus = 'idle'
    } catch (err: any) {
      if (err.status === 409) {
        // 版本冲突 — 保存冲突数据，等待用户决策
        scene.saveStatus = 'conflict'
        scene.conflictData = {
          latestVersion: err.data.latestVersion,
          latestContent: err.data.latestContent
        }
        // 暂停 debounce，不再自动保存，顶部展示冲突横幅
      } else {
        scene.saveStatus = 'error'
      }
    }
  }

  /** 用户选择"保留我的版本" — 用最新版本号强制覆盖 */
  async function resolveConflictKeepMine(sceneId: string) {
    const scene = scenes.value.get(sceneId)
    if (!scene?.conflictData) return
    scene.version = scene.conflictData.latestVersion  // 采用最新版本号
    scene.saveStatus = 'saving'
    await saveScene(sceneId)  // 重新保存（此时版本号已是最新）
    scene.conflictData = undefined
  }

  /** 用户选择"使用最新版本" — 丢弃本地 */
  function resolveConflictUseLatest(sceneId: string) {
    const scene = scenes.value.get(sceneId)
    if (!scene?.conflictData) return
    scene.content = scene.conflictData.latestContent
    scene.version = scene.conflictData.latestVersion
    scene.dirty = false
    scene.saveStatus = 'idle'
    scene.conflictData = undefined
  }

  return { scenes, loadScenes, updateContent, saveScene,
           resolveConflictKeepMine, resolveConflictUseLatest }
})

 * 处理注册、登录、Token 刷新、密码重置全流程
 */
@Service
@Slf4j
@RequiredArgsConstructor
public class AuthService {

    private static final int MAX_LOGIN_ATTEMPTS = 5;
    private static final int LOCK_MINUTES = 15;
    private static final String LOGIN_FAIL_KEY = "login:fail:";
    private static final String REFRESH_TOKEN_KEY = "refresh:";

    private final UserRepository userRepository;
    private final EmailVerificationRepository verificationRepo;
    private final PasswordResetRepository passwordResetRepo;
    private final PasswordEncoder passwordEncoder;   // BCryptPasswordEncoder(strength=12)
    private final JwtTokenProvider jwtProvider;
    private final RedisTemplate<String, String> redisTemplate;
    private final EmailService emailService;

    /**
     * 邮箱注册
     * 1. 邮箱唯一性校验
     * 2. BCrypt 哈希密码
     * 3. 创建用户（emailVerified=false）
     * 4. 生成验证 Token，存 DB，发邮件
     * 5. 签发 JWT 双 Token
     */
    @Transactional
    public AuthResponse register(RegisterRequest req) {
        // 1. 邮箱唯一性
        if (userRepository.existsByEmail(req.getEmail())) {
            throw new BusinessException(ErrorCode.EMAIL_ALREADY_EXISTS, "该邮箱已注册");
        }
        // 2. 密码强度校验（8-32位，含字母和数字）
        validatePasswordStrength(req.getPassword());

        // 3. 创建用户
        User user = User.builder()
            .email(req.getEmail().toLowerCase().trim())
            .passwordHash(passwordEncoder.encode(req.getPassword()))
            .nickname("用户" + RandomUtil.randomNumbers(6))
            .plan("free")
            .emailVerified(false)
            .build();
        userRepository.save(user);

        // 4. 发送验证邮件（异步）
        String verifyToken = UUID.randomUUID().toString();
        verificationRepo.save(EmailVerification.of(user.getId(), verifyToken, 24));
        emailService.sendVerificationEmail(user.getEmail(), verifyToken);

        log.info("[AuthService] 用户注册成功, userId={}, email={}", user.getId(), user.getEmail());
        return buildAuthResponse(user, true);
    }

    /**
     * 邮箱+密码登录
     * 1. 检查账户锁定状态
     * 2. 验证密码
     * 3. 失败记录 Redis，成功清除
     * 4. 签发双 Token
     */
    @Transactional
    public AuthResponse login(LoginRequest req, String clientIp) {
        String lockKey = LOGIN_FAIL_KEY + req.getEmail();

        // 1. 检查锁定
        String failCount = redisTemplate.opsForValue().get(lockKey);
        if (failCount != null && Integer.parseInt(failCount) >= MAX_LOGIN_ATTEMPTS) {
            throw new BusinessException(ErrorCode.ACCOUNT_LOCKED, "账户已锁定，请 15 分钟后重试");
        }

        // 2. 查找用户
        User user = userRepository.findByEmail(req.getEmail().toLowerCase())
            .orElseThrow(() -> {
                incrementLoginFail(lockKey);
                return new BusinessException(ErrorCode.INVALID_CREDENTIALS, "邮箱或密码错误");
            });

        // 3. 验证密码
        if (!passwordEncoder.matches(req.getPassword(), user.getPasswordHash())) {
            int remains = incrementLoginFail(lockKey);
            throw new BusinessException(ErrorCode.INVALID_CREDENTIALS,
                "密码错误，还可尝试 " + (MAX_LOGIN_ATTEMPTS - remains) + " 次");
        }

        // 4. 登录成功，清除失败记录
        redisTemplate.delete(lockKey);
        log.info("[AuthService] 用户登录成功, userId={}, ip={}", user.getId(), clientIp);

        return buildAuthResponse(user, false);
    }

    /**
     * Refresh Token 轮换：旧 Token 换新双 Token
     * 使用 Redis 存储有效 Refresh Token，支持主动吊销
     */
    @Transactional
    public AuthResponse refreshToken(String oldRefreshToken) {
        String redisKey = REFRESH_TOKEN_KEY + oldRefreshToken;
        String userId = redisTemplate.opsForValue().get(redisKey);
        if (userId == null) {
            throw new BusinessException(ErrorCode.INVALID_REFRESH_TOKEN, "登录已过期，请重新登录");
        }
        // 旧 Token 立即失效（防重放）
        redisTemplate.delete(redisKey);

        User user = userRepository.findById(UUID.fromString(userId)).orElseThrow();
        return buildAuthResponse(user, false);
    }

    /** 构建双 Token 响应：生成 Access JWT + Refresh Token 存 Redis */
    private AuthResponse buildAuthResponse(User user, boolean isNew) {
        String accessToken = jwtProvider.generateAccessToken(user);
        String refreshToken = generateAndStoreRefreshToken(user.getId());
        return AuthResponse.builder()
            .accessToken(accessToken)
            .refreshToken(refreshToken)
            .tokenType("Bearer")
            .expiresIn(900)
            .isNew(isNew)
            .emailVerified(user.isEmailVerified())
            .user(UserSummaryDto.from(user))
            .build();
    }

    private String generateAndStoreRefreshToken(UUID userId) {
        String token = HexUtil.encodeHexStr(SecureRandom.getSeed(32));
        redisTemplate.opsForValue().set(
            REFRESH_TOKEN_KEY + token,
            userId.toString(),
            30, TimeUnit.DAYS
        );
        return token;
    }

    private int incrementLoginFail(String lockKey) {
        Long count = redisTemplate.opsForValue().increment(lockKey);
        redisTemplate.expire(lockKey, LOCK_MINUTES, TimeUnit.MINUTES);
        return count.intValue();
    }

    private void validatePasswordStrength(String password) {
        if (password.length() < 8 || password.length() > 32) {
            throw new BusinessException(ErrorCode.WEAK_PASSWORD, "密码长度须在 8–32 位之间");
        }
        if (!password.matches(".*[a-zA-Z].*") || !password.matches(".*\\d.*")) {
            throw new BusinessException(ErrorCode.WEAK_PASSWORD, "密码须同时包含字母和数字");
        }
    }
}

@Service
@Slf4j
@RequiredArgsConstructor
public class PaymentService {

    private final PayOrderRepository orderRepo;
    private final QuotaService quotaService;
    private final UserRepository userRepo;

    // ======================== 创建支付订单 ========================

    /**
     * 创建支付宝 PC 网站支付订单
     * @return 支付宝 form 表单 HTML（前端直接 document.write 提交）
     */
    public String createAlipayPcOrder(UUID userId, String productType, String period) {
        PayOrder order = createOrder(userId, productType, period, "alipay", "pc");

        AlipayTradePagePayRequest request = new AlipayTradePagePayRequest();
        request.setNotifyUrl(alipayConfig.getNotifyUrl());
        request.setReturnUrl(alipayConfig.getReturnUrl());
        request.setBizContent(JSON.toJSONString(Map.of(
            "out_trade_no", order.getOrderNo(),
            "total_amount", String.format("%.2f", order.getAmountFen() / 100.0),
            "subject", getProductTitle(productType, period),
            "product_code", "FAST_INSTANT_TRADE_PAY"
        )));
        return alipayClient.pageExecute(request).getBody();
    }

    /**
     * 创建微信 Native 扫码支付
     * @return 微信支付二维码 URL（前端生成 QR Code 展示）
     */
    public String createWechatNativeOrder(UUID userId, String productType, String period) {
        PayOrder order = createOrder(userId, productType, period, "wechat", "native");

        PrepayRequest req = new PrepayRequest();
        req.setAppid(wechatConfig.getAppId());
        req.setMchid(wechatConfig.getMchId());
        req.setDescription(getProductTitle(productType, period));
        req.setOutTradeNo(order.getOrderNo());
        req.setNotifyUrl(wechatConfig.getNotifyUrl());
        Amount amount = new Amount();
        amount.setTotal(order.getAmountFen());
        req.setAmount(amount);
        PrepayResponse response = nativePayService.prepay(req);
        return response.getCodeUrl();
    }

    // ======================== 支付回调处理 ========================

    /**
     * 支付宝异步通知回调（幂等处理）
     */
    @Transactional
    public void handleAlipayNotify(Map<String, String> params) {
        // 1. 验签（防篡改）
        boolean signValid = AlipaySignature.rsaCheckV1(params,
            alipayConfig.getPublicKey(), "UTF-8", "RSA2");
        if (!signValid) {
            log.error("[PaymentService] 支付宝回调验签失败, params={}", params);
            throw new SecurityException("支付宝签名验证失败");
        }
        // 2. 幂等检查（同一笔订单只处理一次）
        String orderNo = params.get("out_trade_no");
        String tradeStatus = params.get("trade_status");
        PayOrder order = orderRepo.findByOrderNo(orderNo)
            .orElseThrow(() -> new BusinessException("订单不存在: " + orderNo));

        if ("TRADE_SUCCESS".equals(tradeStatus) || "TRADE_FINISHED".equals(tradeStatus)) {
            if ("pending_pay".equals(order.getStatus())) {
                completePayment(order, params.get("trade_no"));
            }
            // 已是 paid/completed 状态时静默忽略（幂等）
        }
    }

    /** 支付完成：更新订单状态 → 充值配额/开通订阅 */
    @Transactional
    private void completePayment(PayOrder order, String tradeNo) {
        orderRepo.updatePaid(order.getId(), tradeNo);
        quotaService.activatePlanOrTopup(order.getUserId(),
            order.getProductType(), order.getPeriod());
        log.info("[PaymentService] 支付成功, orderNo={}, userId={}",
            order.getOrderNo(), order.getUserId());
    }

    private PayOrder createOrder(UUID userId, String productType,
                                  String period, String channel, String scene) {
        int amountFen = PriceConfig.getPrice(productType, period);
        PayOrder order = PayOrder.builder()
            .userId(userId)
            .orderNo(generateOrderNo())
            .payChannel(channel)
            .payScene(scene)
            .productType(productType)
            .period(period)
            .amountFen(amountFen)
            .status("pending_pay")
            .expireAt(Instant.now().plus(30, ChronoUnit.MINUTES))
            .build();
        return orderRepo.save(order);
    }

    private String generateOrderNo() {
        // 格式：YYYYMMDDHHmmss + 8位随机
        return LocalDateTime.now().format(DateTimeFormatter.ofPattern("yyyyMMddHHmmss"))
            + RandomUtil.randomNumbers(8);
    }
}

13.3 支付回调验签实现（V1.4 补充）

安全要求：支付回调接口（/webhook/payment）对外暴露且无需 JWT 鉴权，必须严格验签，否则任何人都可伪造支付成功通知。微信支付使用 RSA 验签，支付宝使用 RSA2 验签，两者均需将公钥配置在环境变量中，严禁硬编码在代码或配置文件里。

微信支付 RSA 验签

/**
 * 微信支付回调接口（V3 API）
 * 微信支付会在回调请求头中携带签名相关字段，使用微信平台公钥证书验证。
 * wechatpay-java SDK 自动处理证书轮换，无需手动管理公钥。
 */
@RestController
@RequestMapping("/api/v1/webhook")
@Slf4j
@RequiredArgsConstructor
public class PaymentWebhookController {

    private final WechatPayService wechatPayService;
    private final AlipayService alipayService;

    /**
     * 微信支付异步通知（V3 API）
     * 验签由 wechatpay-java SDK 在 NotificationParser 中自动完成：
     *   1. 从请求头获取 Wechatpay-Signature、Wechatpay-Timestamp、Wechatpay-Nonce、Wechatpay-Serial
     *   2. 按 "timestamp\nnonce\nbody\n" 格式拼接验签串
     *   3. 用微信平台证书（通过 Wechatpay-Serial 从证书管理器获取）的公钥 RSA 验证 Base64 签名
     *   4. 校验 timestamp 与当前时间差不超过 5 分钟（防重放）
     */
    @PostMapping("/wechat")
    public ResponseEntity<Map<String, String>> handleWechatNotify(
            HttpServletRequest request,
            @RequestBody String rawBody) {

        // 1. 组装 RequestParam 供 SDK 验签
        RequestParam requestParam = new RequestParam.Builder()
            .serialNumber(request.getHeader("Wechatpay-Serial"))
            .nonce(request.getHeader("Wechatpay-Nonce"))
            .signature(request.getHeader("Wechatpay-Signature"))
            .timestamp(request.getHeader("Wechatpay-Timestamp"))
            .body(rawBody)
            .build();

        try {
            // 2. SDK 内部完成 RSA 验签 + 解密通知体（AES-256-GCM，密钥为 apiV3Key）
            Transaction transaction = wechatPayService.parsePayNotify(requestParam);

            // 3. 幂等处理支付结果
            if ("SUCCESS".equals(transaction.getTradeState())) {
                paymentService.handleWechatPaySuccess(
                    transaction.getOutTradeNo(),
                    transaction.getTransactionId()
                );
            }
            // 4. 必须返回 200 + code:SUCCESS，否则微信会重试
            return ResponseEntity.ok(Map.of("code", "SUCCESS", "message", "ok"));

        } catch (ValidationException e) {
            // 验签失败：记录日志并返回 401（微信不会重试）
            log.error("[Webhook] 微信支付验签失败, error={}", e.getMessage());
            return ResponseEntity.status(HttpStatus.UNAUTHORIZED)
                .body(Map.of("code", "FAIL", "message", "验签失败"));
        }
    }

    /**
     * 支付宝异步通知（RSA2 + charset=UTF-8）
     * 支付宝验签使用支付宝公钥（非应用私钥），公钥从支付宝开放平台获取后存入环境变量。
     * 验签步骤：
     *   1. 提取 sign 和 sign_type 参数
     *   2. 对剩余参数按 key 字典升序排列，拼接为 key=value&... 格式
     *   3. 使用支付宝公钥 + SHA256WithRSA 验证 sign（Base64 解码后）
     *   4. 校验 notify_time 与当前时间差不超过 5 分钟
     */
    @PostMapping("/alipay")
    public ResponseEntity<String> handleAlipayNotify(
            @RequestParam Map<String, String> params) {

        log.info("[Webhook] 收到支付宝回调, outTradeNo={}", params.get("out_trade_no"));

        try {
            // 1. SDK 验签（AlipaySignature.rsaCheckV1 使用 RSA2/SHA256WithRSA）
            boolean signVerified = AlipaySignature.rsaCheckV1(
                params,
                alipayConfig.getPublicKey(),    // 支付宝公钥（非 App 私钥）
                "UTF-8",
                "RSA2"
            );

            if (!signVerified) {
                log.error("[Webhook] 支付宝验签失败, params={}", params);
                return ResponseEntity.ok("fail");   // 支付宝要求返回 "fail" 触发重试
            }

            // 2. 幂等处理（TRADE_SUCCESS / TRADE_FINISHED 均为支付成功）
            String tradeStatus = params.get("trade_status");
            if ("TRADE_SUCCESS".equals(tradeStatus) || "TRADE_FINISHED".equals(tradeStatus)) {
                paymentService.handleAlipaySuccess(
                    params.get("out_trade_no"),
                    params.get("trade_no")
                );
            }
            // 3. 返回 "success" 通知支付宝停止重试
            return ResponseEntity.ok("success");

        } catch (AlipayApiException e) {
            log.error("[Webhook] 支付宝验签异常", e);
            return ResponseEntity.ok("fail");
        }
    }
}

/**
 * 内容安全过滤服务
 *
 * 使用 Aho-Corasick 多模式字符串匹配，构建时间 O(N)，查询时间 O(M+K)
 * 敏感词库从 classpath:sensitive-words.txt 加载，支持热更新
 */
@Service
@Slf4j
public class ContentSafetyService {

    /** 敏感词替换字符 */
    private static final char REPLACE_CHAR = '*';

    /** 滑动缓冲区大小（流式过滤，防止敏感词跨 token 被截断） */
    private static final int STREAM_BUFFER_SIZE = 20;

    private volatile Trie sensitiveWordTrie;

    @PostConstruct
    public void init() {
        reloadSensitiveWords();
    }

    /** 热更新：管理员上传新词库后调用 */
    public synchronized void reloadSensitiveWords() {
        Trie.TrieBuilder builder = Trie.builder()
            .ignoreCase()          // 大小写不敏感
            .ignoreOverlaps();     // 只报告非重叠匹配
        loadWordsFromFile().forEach(builder::addKeyword);
        this.sensitiveWordTrie = builder.build();
        log.info("[ContentSafety] 敏感词库已加载");
    }

    /**
     * 过滤输入文本（同步，用于非流式场景）
     * @return FilterResult 包含过滤后文本 + 是否有命中
     */
    public FilterResult filter(String text) {
        if (text == null || text.isBlank()) {
            return FilterResult.clean(text);
        }
        Collection<Emit> emits = sensitiveWordTrie.parseText(text);
        if (emits.isEmpty()) {
            return FilterResult.clean(text);
        }
        // 将命中位置替换为 ***
        char[] chars = text.toCharArray();
        emits.forEach(e -> {
            for (int i = e.getStart(); i <= e.getEnd(); i++) {
                chars[i] = REPLACE_CHAR;
            }
        });
        String filtered = new String(chars);
        log.debug("[ContentSafety] 过滤命中 {} 处", emits.size());
        return FilterResult.dirty(filtered, emits.size());
    }

    /**
     * 流式过滤处理器（SSE 续写场景）
     *
     * 使用滑动缓冲区：累积 STREAM_BUFFER_SIZE 个字符后再过滤输出，
     * 防止敏感词恰好跨越两个 token 边界而漏检。
     */
    public StreamingFilterHandler createStreamingHandler(
            Consumer<String> safeTokenConsumer) {
        return new StreamingFilterHandler(sensitiveWordTrie,
            STREAM_BUFFER_SIZE, safeTokenConsumer);
    }

    private List<String> loadWordsFromFile() {
        try (InputStream is = getClass()
                .getResourceAsStream("/sensitive-words.txt")) {
            return IOUtils.readLines(is, StandardCharsets.UTF_8)
                .stream()
                .filter(line -> !line.isBlank() && !line.startsWith("#"))
                .collect(Collectors.toList());
        } catch (IOException e) {
            log.error("[ContentSafety] 加载敏感词库失败", e);
            return Collections.emptyList();
        }
    }

    @Data
    public static class FilterResult {
        private final String text;
        private final boolean hasViolation;
        private final int violationCount;

        public static FilterResult clean(String text) {
            return new FilterResult(text, false, 0);
        }
        public static FilterResult dirty(String text, int count) {
            return new FilterResult(text, true, count);
        }
    }
}

/**
 * 流式过滤处理器
 * 使用者：SSE onNext 回调中调用 accept(token)，流结束时调用 flush()
 */
public class StreamingFilterHandler {

    private final Trie trie;
    private final int bufferSize;
    private final Consumer<String> safeConsumer;
    private final StringBuilder buffer = new StringBuilder();

    public void accept(String token) {
        buffer.append(token);
        // 缓冲区超过阈值时，过滤并输出安全前缀
        if (buffer.length() >= bufferSize) {
            String safe = flushBuffer(false);
            if (!safe.isEmpty()) safeConsumer.accept(safe);
        }
    }

    public void flush() {
        // 流结束，输出剩余缓冲
        String safe = flushBuffer(true);
        if (!safe.isEmpty()) safeConsumer.accept(safe);
    }

    private String flushBuffer(boolean all) {
        // 过滤 buffer，保留末尾 bufferSize/2 作为下一轮重叠
        int outputLen = all ? buffer.length() : buffer.length() - bufferSize / 2;
        if (outputLen <= 0) return "";
        String toFilter = buffer.substring(0, outputLen);
        buffer.delete(0, outputLen);
        return filterText(toFilter);
    }

    private String filterText(String text) {
        char[] chars = text.toCharArray();
        trie.parseText(text).forEach(e -> {
            for (int i = e.getStart(); i <= e.getEnd(); i++) chars[i] = '*';
        });
        return new String(chars);
    }
}

/**
 * Final Draft (.fdx) XML 导出
 * 遵循 Final Draft 8 FDX 格式规范
 */
@Service
public class ScriptFdxExporter {

    public byte[] export(ScriptDraft draft, List<ScriptScene> scenes) {
        StringBuilder xml = new StringBuilder();
        xml.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
        xml.append("<FinalDraft DocumentType=\"Script\" Template=\"No\" Version=\"1\">\n");
        xml.append("<Content>\n");

        for (ScriptScene scene : scenes) {
            // 场景标题
            xml.append("  <Paragraph Type=\"Scene Heading\">\n");
            xml.append("    <Text>").append(escapeXml(buildSceneHeader(scene))).append("</Text>\n");
            xml.append("  </Paragraph>\n");

            // 解析场景内容
            parseAndWriteSceneContent(xml, scene.getContent());
        }

        xml.append("</Content>\n");
        xml.append("</FinalDraft>\n");
        return xml.toString().getBytes(StandardCharsets.UTF_8);
    }

    private void parseAndWriteSceneContent(StringBuilder xml, String content) {
        String[] lines = content.split("\n");
        boolean prevWasCharacter = false;
        for (String line : lines) {
            if (line.isBlank()) continue;
            String type = detectLineType(line, prevWasCharacter);
            xml.append("  <Paragraph Type=\"").append(type).append("\">\n");
            xml.append("    <Text>").append(escapeXml(line.trim())).append("</Text>\n");
            xml.append("  </Paragraph>\n");
            prevWasCharacter = "Character".equals(type);
        }
    }

    private String detectLineType(String line, boolean prevWasCharacter) {
        // 全大写或中文人名单行 → Character
        if (line.trim().matches("^[\\u4e00-\\u9fa5A-Z·]{1,15}$")) return "Character";
        // 上一行是人名 → Dialogue
        if (prevWasCharacter) return "Dialogue";
        // 括号内容 → Parenthetical
        if (line.trim().startsWith("(") || line.trim().startsWith("（")) return "Parenthetical";
        // 其余 → Action
        return "Action";
    }

    private String buildSceneHeader(ScriptScene scene) {
        return String.format("%s %s - %s",
            scene.getIsInterior() ? "INT." : "EXT.",
            scene.getLocation().toUpperCase(),
            scene.getTimeDesc().toUpperCase());
    }

    private String escapeXml(String text) {
        return text.replace("&", "&").replace("<", "<")
                   .replace(">", ">").replace("\"", """);
    }
}

            progressService.step(taskId, 3, "上传至存储...");
            String filename = "wenshu_export_" +
                LocalDate.now().format(DateTimeFormatter.ofPattern("yyyyMMdd")) + ".zip";
            String downloadUrl = fileStorage.uploadExport(userId, filename, baos.toByteArray());

            progressService.complete(taskId, null);
            log.info("[DataExport] 导出完成, userId={}, url={}", userId, downloadUrl);
            return CompletableFuture.completedFuture(downloadUrl);

        } catch (Exception e) {
            progressService.fail(taskId, "导出失败：" + e.getMessage());
            log.error("[DataExport] 导出失败, userId={}", userId, e);
            return CompletableFuture.failedFuture(e);
        }
    }

    private void exportNovels(UUID userId, ZipOutputStream zos) throws IOException {
        List<Project> projects = projectRepo.findByUserId(userId);
        for (Project p : projects) {
            String base = "novels/" + sanitizeFileName(p.getTitle()) + "/";

            // 大纲 JSON
            writeZipEntry(zos, base + "outline.json",
                JSON.toJSONBytes(buildOutlineDto(p.getId())));

            // 角色档案 JSON
            writeZipEntry(zos, base + "characters.json",
                JSON.toJSONBytes(characterRepo.findByProjectId(p.getId())));

            // 章节正文（每章一个 txt）
            List<Chapter> chapters = chapterRepo.findByProjectId(p.getId());
            for (Chapter c : chapters) {
                String filename = String.format("chapters/chap_%03d_%s.txt",
                    c.getSortOrder(), sanitizeFileName(c.getTitle()));
                writeZipEntry(zos, base + filename,
                    c.getContent().getBytes(StandardCharsets.UTF_8));
            }
        }
    }

    private void writeZipEntry(ZipOutputStream zos, String path, byte[] data)
            throws IOException {
        ZipEntry entry = new ZipEntry(path);
        entry.setSize(data.length);
        zos.putNextEntry(entry);
        zos.write(data);
        zos.closeEntry();
    }

    private String sanitizeFileName(String name) {
        // 移除文件名非法字符
        return name.replaceAll("[\\\\/:*?\"<>|]", "_").trim();
    }

18. 前端 SSE 断连重连策略

// composables/useStreaming.ts（完整版，含断连重连）
// 规则：最多重连 3 次，指数退避（1s / 2s / 4s），超过后提示用户手动重试

import { ref, onUnmounted } from 'vue'

interface StreamingOptions {
  onToken:   (token: string) => void
  onDone:    () => void
  onError:   (err: Error) => void
  onReconnecting?: (attempt: number) => void  // 重连提示回调
}

export function useStreaming(options: StreamingOptions) {
  const isStreaming   = ref(false)
  const reconnectAttempts = ref(0)
  const MAX_RECONNECT = 3

  let eventSource: EventSource | null = null
  let reconnectTimer: ReturnType<typeof setTimeout> | null = null

  // ── 存储最后一次 SSE URL，断连后重连用 ──
  let lastUrl = ''

  /**
   * 开始 SSE 流，自动处理断连重连
   * @param url  SSE 端点 URL（含 query params）
   */
  function startStream(url: string) {
    stop()  // 先清理上一次
    lastUrl = url
    reconnectAttempts.value = 0
    connect(url)
  }

  function connect(url: string) {
    eventSource = new EventSource(url)
    isStreaming.value = true

    eventSource.onmessage = (e) => {
      if (e.data === '[DONE]') {
        stop()
        options.onDone()
        return
      }
      reconnectAttempts.value = 0  // 收到数据，重置重连计数
      options.onToken(e.data)
    }

    eventSource.addEventListener('error', () => {
      // EventSource 内部已重试多次后才触发 error，说明连接确实中断
      if (eventSource?.readyState === EventSource.CLOSED) {
        handleDisconnect()
      }
    })

    // 自定义完成事件（服务端发送 event: done）
    eventSource.addEventListener('done', () => {
      stop()
      options.onDone()
    })
  }

  function handleDisconnect() {
    if (reconnectAttempts.value >= MAX_RECONNECT) {
      stop()
      options.onError(new Error('网络连接中断，请重试'))
      return
    }
    reconnectAttempts.value++
    const delay = Math.pow(2, reconnectAttempts.value - 1) * 1000  // 1s, 2s, 4s
    options.onReconnecting?.(reconnectAttempts.value)

    reconnectTimer = setTimeout(() => {
      connect(lastUrl)  // 重连（SSE 会从头开始，服务端需幂等）
    }, delay)
  }

  /** 主动停止 */
  function stop() {
    clearTimeout(reconnectTimer!)
    eventSource?.close()
    eventSource = null
    isStreaming.value = false
  }

  onUnmounted(stop)

  return { isStreaming, reconnectAttempts, startStream, stop }
}

// extensions/AiGeneratedMark.ts
// 自定义 TipTap Mark，标记 AI 生成的文本范围
// 渲染为带 data-ai 属性的 span，CSS 控制左侧绿线和悬浮操作栏

import { Mark, mergeAttributes } from '@tiptap/core'

export interface AiGeneratedOptions {
  HTMLAttributes: Record<string, unknown>
}

declare module '@tiptap/core' {
  interface Commands<ReturnType> {
    aiGenerated: {
      /**
       * 将选区标记为 AI 生成内容
       * @param attrs  generatedAt: 生成时间戳; sessionId: 本次续写的唯一 ID
       */
      setAiGenerated: (attrs: { generatedAt: number; sessionId: string }) => ReturnType
      /** 接受 AI 内容（移除标记，内容保留） */
      acceptAiGenerated: (sessionId: string) => ReturnType
      /** 拒绝 AI 内容（移除标记和内容） */
      rejectAiGenerated: (sessionId: string) => ReturnType
    }
  }
}

export const AiGeneratedMark = Mark.create<AiGeneratedOptions>({
  name: 'aiGenerated',

  addOptions() {
    return { HTMLAttributes: {} }
  },

  // 标记属性：生成时间戳 + 本次续写 sessionId
  addAttributes() {
    return {
      generatedAt: {
        default: null,
        parseHTML: el => el.getAttribute('data-generated-at'),
        renderHTML: attrs => ({ 'data-generated-at': attrs.generatedAt }),
      },
      sessionId: {
        default: null,
        parseHTML: el => el.getAttribute('data-session-id'),
        renderHTML: attrs => ({ 'data-session-id': attrs.sessionId }),
      },
    }
  },

  // 渲染为 span，外层 CSS 通过 [data-ai-generated] 选择器绑定样式
  parseHTML() {
    return [{ tag: 'span[data-ai-generated]' }]
  },
  renderHTML({ HTMLAttributes }) {
    return ['span', mergeAttributes(
      this.options.HTMLAttributes,
      HTMLAttributes,
      { 'data-ai-generated': '' }
    ), 0]
  },

  addCommands() {
    return {
      setAiGenerated: (attrs) => ({ commands }) => {
        return commands.setMark(this.name, attrs)
      },

      acceptAiGenerated: (sessionId) => ({ tr, state, dispatch }) => {
        // 遍历文档，移除匹配 sessionId 的所有 AiGenerated mark（内容保留）
        state.doc.descendants((node, pos) => {
          node.marks.forEach(mark => {
            if (mark.type.name === 'aiGenerated' &&
                mark.attrs.sessionId === sessionId) {
              tr.removeMark(pos, pos + node.nodeSize, mark.type)
            }
          })
        })
        if (dispatch) dispatch(tr)
        return true
      },

      rejectAiGenerated: (sessionId) => ({ tr, state, dispatch }) => {
        // 遍历文档，删除匹配 sessionId 的节点内容
        const ranges: Array<{ from: number; to: number }> = []
        state.doc.descendants((node, pos) => {
          if (node.marks.some(m =>
            m.type.name === 'aiGenerated' &&
            m.attrs.sessionId === sessionId)) {
            ranges.push({ from: pos, to: pos + node.nodeSize })
          }
        })
        // 从后往前删除，避免位置偏移
        ranges.reverse().forEach(r => tr.delete(r.from, r.to))
        if (dispatch) dispatch(tr)
        return true
      },
    }
  },
})

19.2 SSE Token 流增量插入 TipTap

续写时 LLM 以 token 流推送文本，需要在 TipTap 文档中增量追加，并在流式过程中给追加内容自动添加 AiGeneratedMark。

// composables/useAiWriter.ts
// SSE 流式续写：将 token 逐步插入 TipTap，并自动加 AiGenerated 标记

import { ref } from 'vue'
import type { Editor } from '@tiptap/core'
import { useStreaming } from './useStreaming'
import { continueWriting } from '@/api/novel'
import { useNovelStore } from '@/stores/novel'

export function useAiWriter(editor: Ref<Editor | null>) {
  const novelStore = useNovelStore()
  const sessionId  = ref('')      // 本次续写的唯一 ID，用于标记和管理
  const insertFrom = ref(0)       // 记录本次续写开始插入的位置

  const { isStreaming, startStream, stop } = useStreaming({
    onToken: (token: string) => {
      if (!editor.value) return
      const { state, commands } = editor.value

      // 在光标末尾插入 token，同时加上 AiGeneratedMark
      editor.value
        .chain()
        .focus()
        .insertContent({
          type: 'text',
          text: token,
          marks: [{
            type: 'aiGenerated',
            attrs: {
              generatedAt: Date.now(),
              sessionId: sessionId.value
            }
          }]
        })
        .run()
    },

    onDone: () => {
      // 流式结束：通知 Pinia store，前端展示「接受 / 重写」操作栏
      novelStore.setPendingAiContent(sessionId.value)
    },

    onError: (err) => {
      // 生成失败：回滚插入的内容
      if (editor.value && insertFrom.value > 0) {
        const endPos = editor.value.state.doc.content.size
        editor.value.chain()
          .deleteRange({ from: insertFrom.value, to: endPos })
          .run()
      }
      console.error('[AiWriter] SSE 错误:', err)
    },

    onReconnecting: (attempt) => {
      console.warn(`[AiWriter] 重连第 ${attempt} 次`)
    }
  })

  /**
   * 发起 AI 续写
   * @param chapterId   章节 ID
   * @param instruction 用户附加指令（可选）
   */
  function startContinuation(chapterId: string, instruction?: string) {
    // 记录插入起始位置（用于失败回滚）
    insertFrom.value = editor.value?.state.doc.content.size ?? 0
    sessionId.value  = `session-${Date.now()}-${Math.random().toString(36).slice(2)}`

    const source = continueWriting(chapterId, instruction)
    startStream(source)
  }

  /** 接受 AI 内容：移除标记，内容融入正文 */
  function acceptContent() {
    editor.value?.commands.acceptAiGenerated(sessionId.value)
    novelStore.clearPendingAiContent()
    // 通知后端创建版本快照（异步，不阻塞 UI）
    novelStore.triggerSnapshot()
  }

  /** 拒绝 AI 内容：删除本次生成的所有内容 */
  function rejectContent() {
    editor.value?.commands.rejectAiGenerated(sessionId.value)
    novelStore.clearPendingAiContent()
  }

  return { isStreaming, startContinuation, acceptContent, rejectContent, stop }
}

/**
 * 将 diff-match-patch 输出的 diffs 数组转换为 HTML 字符串
 * DIFF_DELETE(-1)：红底删除线
 * DIFF_INSERT(+1)：绿底
 * DIFF_EQUAL(0)：普通文本
 */
function diffsToHtml(diffs: [number, string][]): string {
  return diffs.map(([op, text]) => {
    const escaped = text
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/\n/g, '<br>')
    if (op === DiffMatchPatch.DIFF_DELETE)
      return `<del style="background:#fecaca;color:#991b1b;text-decoration:line-through;
                         border-radius:2px;padding:0 2px;">${escaped}</del>`
    if (op === DiffMatchPatch.DIFF_INSERT)
      return `<ins style="background:#bbf7d0;color:#166534;text-decoration:none;
                         border-radius:2px;padding:0 2px;">${escaped}</ins>`
    return escaped
  }).join('')
}

// 字符级 diff（适合中文，比行级更细致）

  const diffs = dmp.diff_main(props.snapshotContent, props.currentContent)
  dmp.diff_cleanupSemantic(diffs)  // 语义清理，减少噪音
  return diffsToHtml(diffs)
})

    if (op === DiffMatchPatch.DIFF_DELETE) removed += text.length
  })
  return { added, removed }
})

关键设计决策
• Mark vs Node：AI 内容标记选用 Mark 而非 Node，原因是用户可能选中 AI 内容后继续手动编辑，Mark 允许部分接受（光标在 AI 内容中间继续打字时，该位置之前的 mark 可单独接受）。
• sessionId 隔离：每次续写生成唯一 sessionId，支持同一章节存在多段未接受的 AI 内容，互不干扰。
• debounce 保存：编辑区内容变更通过 debounce 500ms 同步到 Pinia store，store 再 debounce 2s 调用自动保存 API，避免高频写请求。

// ChapterSearchService.java
@Service
@Slf4j
@RequiredArgsConstructor
public class ChapterSearchService {

    private final ChapterMapper chapterMapper;
    private static final int SNIPPET_CONTEXT = 30;  // 关键词前后各30字
    private static final int MAX_SNIPPETS_PER_CHAPTER = 5; // 每章最多返回5处，防止超长响应

    /**
     * 全书关键词搜索，返回按章节分组的结果。
     *
     * @param projectId     作品 ID
     * @param keyword       搜索关键词（非空，长度 1–50）
     * @param caseSensitive 是否区分大小写（默认 false）
     * @param wholeWord     是否全词匹配（默认 false）
     * @return 按章节排序的匹配结果列表
     */
    public List<ChapterSearchResult> search(UUID projectId, String keyword,
                                             boolean caseSensitive, boolean wholeWord) {
        log.info("[Search] 全书搜索开始, projectId={}, keyword={}", projectId, keyword);

        // 1. 查询有匹配的章节（SQL 层过滤）
        List<ChapterMatchSummary> matchedChapters =
            chapterMapper.findChaptersContaining(projectId, keyword, caseSensitive);

        if (matchedChapters.isEmpty()) {
            return Collections.emptyList();
        }

        // 2. 对每个匹配章节提取上下文片段（Java 层处理，避免复杂 SQL）
        return matchedChapters.stream()
            .map(c -> buildResult(c, keyword, caseSensitive, wholeWord))
            .collect(toList());
    }

    private ChapterSearchResult buildResult(ChapterMatchSummary chapter,
                                             String keyword, boolean cs, boolean wholeWord) {
        String content = chapter.getContent();
        String searchTarget = cs ? content : content.toLowerCase();
        String searchKey = cs ? keyword : keyword.toLowerCase();

        List<SnippetDto> snippets = new ArrayList<>();
        int pos = 0;
        while (snippets.size() < MAX_SNIPPETS_PER_CHAPTER) {
            int idx = searchTarget.indexOf(searchKey, pos);
            if (idx == -1) break;

            if (!wholeWord || isWordBoundary(content, idx, keyword.length())) {
                int start = Math.max(0, idx - SNIPPET_CONTEXT);
                int end = Math.min(content.length(), idx + keyword.length() + SNIPPET_CONTEXT);
                snippets.add(new SnippetDto(content.substring(start, end), idx - start, idx - start + keyword.length()));
            }
            pos = idx + 1;
        }

        return ChapterSearchResult.builder()
            .chapterId(chapter.getId())
            .chapterTitle(chapter.getTitle())
            .sortOrder(chapter.getSortOrder())
            .matchCount(chapter.getMatchCount())
            .snippets(snippets)
            .build();
    }

    private boolean isWordBoundary(String text, int idx, int len) {
        boolean leftOk  = idx == 0 || !Character.isLetterOrDigit(text.charAt(idx - 1));
        boolean rightOk = idx + len >= text.length() || !Character.isLetterOrDigit(text.charAt(idx + len));
        return leftOk && rightOk;
    }
}
<!-- ChapterMapper.xml — 全书搜索 SQL -->
<select id="findChaptersContaining" resultType="ChapterMatchSummary">
    SELECT
        c.id,
        c.title,
        c.sort_order,
        c.content,
        v.sort_order AS volume_sort_order,
        /* 估算匹配数：(原文长度 - 替换后长度) / 关键词长度 */
        (LENGTH(LOWER(c.content)) - LENGTH(REPLACE(LOWER(c.content), LOWER(#{keyword}), '')))
            / GREATEST(LENGTH(#{keyword}), 1) AS match_count
    FROM chapters c
    JOIN volumes v ON c.volume_id = v.id
    WHERE c.project_id = #{projectId}
      AND (
        #{caseSensitive} = true  AND c.content LIKE '%' || #{keyword} || '%'
        OR
        #{caseSensitive} = false AND LOWER(c.content) LIKE LOWER('%' || #{keyword} || '%')
      )
      AND c.content IS NOT NULL
      AND LENGTH(#{keyword}) BETWEEN 1 AND 50  -- 防止空字符串或超长关键词
    ORDER BY v.sort_order, c.sort_order
</select>

/**
 * 批量替换服务（事务保护，失败全部回滚）
 */
@Service
@Slf4j
@RequiredArgsConstructor
public class SearchReplaceService {

    private final ChapterRepository chapterRepo;
    private final CharacterRepository characterRepo;
    private final SnapshotService snapshotService;

    /**
     * 全书批量替换，操作前自动创建版本快照。
     *
     * @param projectId              作品 ID
     * @param search                 被替换词
     * @param replace                替换词
     * @param chapterIds             指定章节范围（空=全部）
     * @param caseSensitive          是否区分大小写
     * @param updateCharacterProfile 是否联动更新角色档案名称
     * @return 替换结果摘要
     */
    @Transactional
    public ReplaceResult replaceAll(UUID projectId, String search, String replace,
                                     List<UUID> chapterIds, boolean caseSensitive,
                                     boolean updateCharacterProfile) {
        // 1. 先查询受影响章节（不修改），用于创建快照和统计
        List<Chapter> targets = chapterIds.isEmpty()
            ? chapterRepo.findByProjectId(projectId)
            : chapterRepo.findByIds(chapterIds);

        List<Chapter> affected = targets.stream()
            .filter(c -> containsKeyword(c.getContent(), search, caseSensitive))
            .toList();

        if (affected.isEmpty()) {
            return ReplaceResult.empty();
        }

        // 2. 批量创建版本快照（类型 auto_before_replace，在修改前保护原文）
        UUID snapshotId = snapshotService.createBatchSnapshot(
            affected.stream().map(Chapter::getId).toList(), "auto_before_replace");

        // 3. 执行替换
        int totalReplaced = 0;
        for (Chapter chapter : affected) {
            String newContent = doReplace(chapter.getContent(), search, replace, caseSensitive);
            int count = countOccurrences(chapter.getContent(), search, caseSensitive);
            chapterRepo.updateContent(chapter.getId(), newContent);
            totalReplaced += count;
            log.debug("[Replace] 章节替换完成, chapterId={}, count={}", chapter.getId(), count);
        }

        // 4. 联动更新角色档案（可选）
        if (updateCharacterProfile) {
            updateCharacterNames(projectId, search, replace);
        }

        log.info("[Replace] 全书替换完成, projectId={}, search={}, replace={}, total={}",
            projectId, search, replace, totalReplaced);

        return new ReplaceResult(totalReplaced, affected.size(), snapshotId);
    }

    private String doReplace(String content, String search, String replace, boolean cs) {
        return cs ? content.replace(search, replace)
                  : content.replaceAll("(?i)" + Pattern.quote(search), Matcher.quoteReplacement(replace));
    }
}

21.3 角色名联动更新

/**
 * 角色名联动更新：当替换词命中角色名时，自动更新角色档案。
 * 在 SearchReplaceService.replaceAll() 中 updateCharacterProfile=true 时触发。
 */
private void updateCharacterNames(UUID projectId, String oldName, String newName) {
    List<Character> matched = characterRepo.findByProjectId(projectId)
        .stream()
        .filter(c -> c.getName().equals(oldName))
        .toList();

    if (matched.isEmpty()) return;

    matched.forEach(c -> {
        c.setName(newName);
        // 同步更新别名列表（如 ["小天","天哥"] 不变，仅主名更新）
        characterRepo.update(c);
        log.info("[Replace] 角色档案同步更新, characterId={}, {} → {}", c.getId(), oldName, newName);
    });
}

22.1 数据采集时机

写作统计采用差量采集策略：在每次章节内容保存时（PUT /chapters/{id}）比较旧字数与新字数，将净增量分类写入 writing_daily_stats。不使用定时快照，保证数据准确且无性能负担。 
/**
 * 写作统计采集服务（在 ChapterService.updateChapter 中调用）
 *
 * 采集规则：
 *  - 用户手动保存（非 AI 触发）→ 净增量计入 manual_chars
 *  - 用户接受 AI 续写内容（isAiContent=true）→ 计入 ai_accepted_chars
 *  - 字数净减少（删除内容）→ 两个字段均不增加（不记录负数）
 *  - 全局行（project_id=NULL）与作品行同步更新
 */
@Service
@Slf4j
@RequiredArgsConstructor
public class WritingStatsService {

    private final WritingDailyStatsMapper statsMapper;

    /**
     * 记录一次字数变化（由 ChapterService 在保存章节后调用）
     *
     * @param userId      用户 ID
     * @param projectId   作品 ID
     * @param charDelta   字数净增量（负数则忽略）
     * @param isAiContent 是否为接受 AI 生成的内容
     */
    @Async("statsExecutor")
    @Transactional(propagation = Propagation.REQUIRES_NEW)
    public void record(UUID userId, UUID projectId, int charDelta, boolean isAiContent) {
        if (charDelta <= 0) return;   // 删除操作不计入统计

        LocalDate today = LocalDate.now();
        int currentHour = LocalTime.now().getHour();

        // 更新作品行（project_id = projectId）
        upsertStats(userId, projectId, today, charDelta, isAiContent, currentHour);
        // 更新全局行（project_id = NULL，用于跨作品总计）
        upsertStats(userId, null, today, charDelta, isAiContent, currentHour);

        log.debug("[WritingStats] 字数记录, userId={}, projectId={}, delta={}, ai={}",
            userId, projectId, charDelta, isAiContent);
    }

    private void upsertStats(UUID userId, UUID projectId, LocalDate date,
                              int delta, boolean isAi, int hour) {
        statsMapper.upsert(userId, projectId, date, delta, isAi, hour);
        // SQL: INSERT ... ON CONFLICT(user_id, project_id, stat_date) DO UPDATE

        //      SET ai_accepted_chars = ai_accepted_chars + #{delta} WHERE #{isAi} = true
        //      SET total_chars = total_chars + #{delta}
        //      SET peak_hour = CASE WHEN new_total > old_total THEN #{hour} ELSE peak_hour END
    }
}

// ChapterService.updateChapter 中的调用点（关键集成位置）
@Transactional
public Chapter updateChapter(UUID chapterId, UpdateChapterRequest req, UUID userId) {
    Chapter old = chapterRepo.findById(chapterId).orElseThrow();
    int oldWordCount = old.getWordCount();

    // 更新章节内容
    old.setContent(req.getContent());
    int newWordCount = TextCountUtil.countChinese(req.getContent());
    old.setWordCount(newWordCount);
    chapterRepo.save(old);

    // 异步记录统计（不阻塞保存响应）
    int delta = newWordCount - oldWordCount;
    writingStatsService.record(userId, old.getProjectId(), delta, req.isAiContent());

    return old;
}

<select id="getHeatmap" resultType="HeatmapCell">
    /* 热力图：按 星期几(0-6) × 小时(0-23) 聚合近N天的写作字数 */
    SELECT
        EXTRACT(DOW FROM stat_date)::INT  AS day_of_week,  -- 0=周日, 1=周一, ..., 6=周六
        peak_hour                          AS hour,
        SUM(total_chars)                   AS total_chars
    FROM writing_daily_stats
    WHERE user_id  = #{userId}
      AND project_id IS NULL              -- 全局统计行
      AND stat_date >= CURRENT_DATE - INTERVAL '#{days} days'
      AND total_chars > 0
    GROUP BY EXTRACT(DOW FROM stat_date), peak_hour
    ORDER BY day_of_week, hour
</select>

<!-- 连续日更天数查询（利用 PostgreSQL 窗口函数）-->
<select id="getStreak" resultType="StreakDto">
    WITH daily AS (
        SELECT stat_date,
               total_chars,
               (total_chars >= #{goalChars})::INT AS reached_goal
        FROM writing_daily_stats
        WHERE user_id = #{userId} AND project_id IS NULL
        ORDER BY stat_date DESC
    ),
    consecutive AS (
        SELECT stat_date,
               reached_goal,
               ROW_NUMBER() OVER (ORDER BY stat_date DESC) -
               ROW_NUMBER() OVER (PARTITION BY reached_goal ORDER BY stat_date DESC) AS grp
        FROM daily
    )
    SELECT
        /* 当前连续 */
        (SELECT COUNT(*) FROM consecutive WHERE reached_goal = 1 AND grp = 0)::INT AS current_streak,
        /* 历史最长 */
        MAX(cnt)::INT AS best_streak
    FROM (
        SELECT COUNT(*) AS cnt
        FROM consecutive
        WHERE reached_goal = 1
        GROUP BY grp
    ) t
</select>
